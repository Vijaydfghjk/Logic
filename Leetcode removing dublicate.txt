
package main

import "fmt"

func removeDuplicates(nums []int) int {

	var a int = 1
	for i := 1; i < len(nums); i++ {

		if nums[i] != nums[i-1] { //i =2

			nums[a] = nums[i]
			a++
		}

	}
	
	fmt.Println(nums[:a])

	return a
}

func main() {

	nums := []int{0, 0, 1, 1, 1, 2, 2, 3, 3, 4}
	value := removeDuplicates(nums)

	fmt.Println(value)
}
